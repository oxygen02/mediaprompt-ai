## 四、前端技术要点

### 4.1 事件绑定规范

**推荐使用 onclick 属性绑定：**

```html
<button onclick="switchLang('zh')">中文</button>
```

```javascript
<script>
  function switchLang(lang) {
    currentLang = lang;
    // ...
  }
  
  // 或在DOM加载后赋值
  document.getElementById('btnZh').onclick = function() {
    switchLang('zh');
  };
</script>
```

**避免使用 addEventListener + querySelectorAll：**
```javascript
// 不推荐
document.querySelectorAll('.lang-btn').forEach(btn => {
  btn.addEventListener('click', ...);
});
```

### 4.2 变量定义规范

**全局变量只定义一次：**
```javascript
// 在script开头定义
var currentLang = 'zh';
var currentType = 'document';

// 不要在后面重复定义
// ❌ let currentLang = 'zh';
```

### 4.3 类别切换逻辑

```javascript
function switchCategory(type) {
  currentType = type;
  
  // 更新按钮样式
  updateCategoryButtons(type);
  
  // 显示/隐藏对应选项区域
  var options = {
    document: 'documentOptions',
    image: 'imageOptions',
    video: 'videoOptions',
    website: 'websiteOptions'
  };
  
  Object.keys(options).forEach(key => {
    var el = document.getElementById(options[key]);
    if (el) {
      if (key === type) el.classList.remove('hidden');
      else el.classList.add('hidden');
    }
  });
  
  // 视频专属选项
  var videoSection = document.getElementById('videoSection');
  if (videoSection) {
    videoSection.style.display = type === 'video' ? 'block' : 'none';
  }
}
```

## 五、后端技术要点

### 5.1 视频压缩流程

```javascript
async function compressVideo(videoPath, maxSize) {
  // 1. 获取视频信息
  const info = await getVideoInfo(videoPath);
  
  // 2. 计算压缩参数
  const targetBitrate = Math.floor((maxSize * 8) / info.duration);
  
  // 3. 使用FFmpeg压缩
  const compressedPath = await ffmpeg.compress(videoPath, {
    videoBitrate: targetBitrate,
    audioBitrate: '128k',
    scale: 'iw*0.75:-1'  // 降分辨率
  });
  
  return compressedPath;
}
```

### 5.2 千问VL调用

```javascript
async function analyzeVideo(videoUrl, options) {
  // 1. 压缩视频
  const compressed = await compressVideo(videoUrl, 10 * 1024 * 1024);
  
  // 2. 构建提示词
  const prompt = buildPrompt(options);
  
  // 3. 调用千问VL
  const response = await qwen.chat.completions.create({
    model: 'qwen3-vl-plus',
    messages: [
      { role: 'system', content: SYSTEM_PROMPT },
      { role: 'user', content: [
        { type: 'text', text: prompt },
        { type: 'image_url', image_url: { url: compressed } }
      ]}
    ]
  });
  
  return response.choices[0].message.content;
}
```
